  <script>
    document.getElementById("contactForm").addEventListener("submit", function(e) {
      e.preventDefault();

      const name = document.getElementById("name").value;
      const devices = parseInt(document.getElementById("devices").value);
      const plan = document.getElementById("plan").value;

      let pricePerDevice = 0;

      if (plan === "monthly") pricePerDevice = 25;
      if (plan === "yearly") pricePerDevice = 250;
      if (plan === "multi") pricePerDevice = 450;

      const total = pricePerDevice * devices;

      document.getElementById("result").innerText =
        `Thank you ${name}! Your estimated cost is $${total}. We will contact you soon.`;
    });
  </script>
